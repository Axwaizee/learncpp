19. Largest number in array

#include <iostream>
using namespace std;

int main() {
    int arr[5];
    cout << "Enter 5 numbers: ";
    for (int i = 0; i < 5; i++)
        cin >> arr[i];

    int largest = arr[0];
    for (int i = 1; i < 5; i++)
        if (arr[i] > largest)
            largest = arr[i];

    cout << "Largest = " << largest;
    return 0;
}

20. Average of 5 numbers

    cout << "Average = " << sum / 5.0;
    return 0;
}

    return 0;
}

8. Simple Practice Projects

27. Simple calculator

#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;
    cout << "Enter first number: ";
    cin >> a;
    cout << "Enter operator (+, -, *, /): ";
    cin >> op;
    cout << "Enter second number: ";
    cin >> b;

    if (op == '+') cout << a + b;
    else if (op == '-') cout << a - b;
    else if (op == '*') cout << a * b;
    else if (op == '/') cout << a / b;
    else cout << "Invalid operator";
    return 0;
}

30. Triangle pattern

#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter number of rows: ";
    cin >> n;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= i; j++)
            cout << "*";
        cout << endl;
    }
    return 0;
}
